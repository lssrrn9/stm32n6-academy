01 I2C: TMP117 + Display OLED SSD1306

Protocolo I2C

I2C (Inter-Integrated Circuit) es un bus serie sincrono multi-maestro/multi-esclavo que usa solo dos lineas:

  • SDA (Serial Data) — linea de datos bidireccional
  • SCL (Serial Clock) — linea de reloj generada por el master

Cada dispositivo esclavo tiene una direccion unica de 7 bits (0x00-0x7F). El master inicia la comunicacion con una condicion START, envia la direccion del esclavo + bit R/W, y luego intercambia datos. La comunicacion termina con una condicion STOP.

Nota: I2C opera en modos Standard (100 kHz), Fast (400 kHz) y Fast+ (1 MHz). En este modulo usamos 100 kHz para maxima compatibilidad.

Sensor TMP117

El TMP117 es un sensor de temperatura digital de alta precision (+/-0.1 C) con interfaz I2C. Su direccion por defecto es 0x48 (7 bits). El registro de temperatura (0x00) devuelve un valor de 16 bits con signo, donde 1 LSB = 0.0078 C.

ParametroValor
Direccion I2C0x48 (7 bits)
Rango de temperatura-55 C a +125 C
Precision+/-0.1 C (typ)
Resolucion0.0078 C/LSB
Registro de temperatura0x00 (16-bit signed)

Display OLED SSD1306

El controlador SSD1306 maneja pantallas OLED de 128x64 pixeles via I2C. Su direccion tipica es 0x3C. Requiere una secuencia de inicializacion de comandos antes de poder mostrar datos.

Codigo HAL: Lectura de temperatura TMP117

C — TMP117_ReadTemperature
#define TMP117_ADDR         (0x48 << 1)   /* Direccion 8-bit para HAL */
#define TMP117_TEMP_REG     0x00
#define TMP117_RESOLUTION   0.0078125f    /* 1 LSB = 7.8125 mili-C */

float TMP117_ReadTemperature(void)
{
  uint8_t reg = TMP117_TEMP_REG;
  uint8_t buf[2];
  int16_t raw;
  float temp;

  /* Escribe el registro a leer */
  HAL_I2C_Master_Transmit(&hi2c1, TMP117_ADDR, &reg, 1, HAL_MAX_DELAY);
  /* Lee 2 bytes de temperatura */
  HAL_I2C_Master_Receive(&hi2c1, TMP117_ADDR, buf, 2, HAL_MAX_DELAY);

  raw = (int16_t)((buf[0] << 8) | buf[1]);
  temp = raw * TMP117_RESOLUTION;

  return temp;
}

Codigo HAL: Mostrar temperatura en OLED

C — SSD1306_ShowTemperature
#include <stdio.h>
#include <string.h>

/* Fuente de 5x7 para digitos (simplificada) */
static const uint8_t font5x7[10][5] = {
  {0x3E,0x51,0x49,0x45,0x3E}, /* 0 */
  {0x00,0x42,0x7F,0x40,0x00}, /* 1 */
  {0x42,0x61,0x51,0x49,0x46}, /* 2 */
  {0x21,0x41,0x45,0x4B,0x31}, /* 3 */
  {0x18,0x14,0x12,0x7F,0x10}, /* 4 */
  {0x27,0x45,0x45,0x45,0x39}, /* 5 */
  {0x3C,0x4A,0x49,0x49,0x30}, /* 6 */
  {0x01,0x71,0x09,0x05,0x03}, /* 7 */
  {0x36,0x49,0x49,0x49,0x36}, /* 8 */
  {0x06,0x49,0x49,0x29,0x1E}, /* 9 */
};

static uint8_t oled_buffer[8][128];  /* 8 paginas x 128 columnas */

void SSD1306_Clear(void)
{
  memset(oled_buffer, 0, sizeof(oled_buffer));
  for (uint8_t page = 0; page < 8; page++)
  {
    SSD1306_WriteCommand(0xB0 + page);
    SSD1306_WriteCommand(0x00);
    SSD1306_WriteCommand(0x10);
    uint8_t data[129];
    data[0] = 0x40; /* Co=0, D/C#=1 (datos) */
    memcpy(&data[1], oled_buffer[page], 128);
    HAL_I2C_Master_Transmit(&hi2c1, SSD1306_ADDR, data, 129, HAL_MAX_DELAY);
  }
}

void SSD1306_DrawChar(uint8_t x, uint8_t page, char c)
{
  if (c >= '0' && c <= '9')
  {
    uint8_t idx = c - '0';
    for (uint8_t i = 0; i < 5; i++)
    {
      if (x + i < 128) oled_buffer[page][x + i] = font5x7[idx][i];
    }
    if (x + 5 < 128) oled_buffer[page][x + 5] = 0x00; /* espacio */
  }
  else if (c == '.')
  {
    if (x < 128) oled_buffer[page][x] = 0x60;
  }
  else if (c == '-')
  {
    if (x < 128) oled_buffer[page][x] = 0x08;
  }
  else if (c == 'C')
  {
    uint8_t fontC[5] = {0x3E,0x41,0x41,0x41,0x22};
    for (uint8_t i = 0; i < 5; i++)
      if (x + i < 128) oled_buffer[page][x + i] = fontC[i];
  }
}

void SSD1306_DrawString(uint8_t x, uint8_t page, const char* str)
{
  while (*str)
  {
    SSD1306_DrawChar(x, page, *str);
    x += 6;
    str++;
  }
}

void SSD1306_Refresh(void)
{
  for (uint8_t page = 0; page < 8; page++)
  {
    SSD1306_WriteCommand(0xB0 + page);
    SSD1306_WriteCommand(0x00);
    SSD1306_WriteCommand(0x10);
    uint8_t data[129];
    data[0] = 0x40;
    memcpy(&data[1], oled_buffer[page], 128);
    HAL_I2C_Master_Transmit(&hi2c1, SSD1306_ADDR, data, 129, HAL_MAX_DELAY);
  }
}

void SSD1306_ShowTemperature(float temp)
{
  char buf[16];
  snprintf(buf, sizeof(buf), "%+.2f C", temp);
  memset(oled_buffer, 0, sizeof(oled_buffer));
  SSD1306_DrawString(10, 2, "TEMP:");
  SSD1306_DrawString(10, 4, buf);
  SSD1306_Refresh();
}

02 SPI: Sensor BMP280 (Presion y Temperatura)

Protocolo SPI

SPI (Serial Peripheral Interface) es un bus sincrono de comunicacion full-duplex que usa 4 senales principales:

  • MOSI (Master Out Slave In) — datos del master al esclavo
  • MISO (Master In Slave Out) — datos del esclavo al master
  • SCK (Serial Clock) — reloj generado por el master
  • CS (Chip Select) — activa el esclavo seleccionado (activo bajo)

SPI opera en diferentes modos (0-3) definidos por CPOL (polaridad del reloj) y CPHA (fase de muestreo). El BMP280 usa modo 0 (CPOL=0, CPHA=0).

Sensor BMP280

El BMP280 es un sensor de presion barometrica y temperatura de Bosch. Mide presion absoluta (300-1100 hPa) con una precision de +/-1 hPa, lo que permite estimar altitud con resolucion de ~1 metro.

ParametroValor
Registro ID0xD0 (valor esperado: 0x58)
Registro de control0xF4
Registro de configuracion0xF5
Datos de presion (raw)0xF7-0xF9 (20 bits)
Datos de temperatura (raw)0xFA-0xFC (20 bits)
Datos de calibracion0x88-0xA1 (26 bytes)

Diagrama de conexiones

STM32N6 BMP280 PA5(SCK)-----------SCK PA6(MISO)----------SDO PA7(MOSI)----------SDI PA4(CS)------------CSB (Chip Select, activo bajo) 3.3V---------------VCC GND----------------GND

  HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);

}

Codigo HAL: Funciones SPI para BMP280

C — BMP280_ReadID y funciones SPI
#define BMP280_CS_PORT   GPIOA
#define BMP280_CS_PIN    GPIO_PIN_4

#define BMP280_REG_ID       0xD0
#define BMP280_REG_CTRL     0xF4
#define BMP280_REG_CONFIG   0xF5
#define BMP280_REG_PRESS    0xF7
#define BMP280_REG_TEMP     0xFA
#define BMP280_REG_CALIB    0x88

static void BMP280_CS_Low(void)  { HAL_GPIO_WritePin(BMP280_CS_PORT, BMP280_CS_PIN, GPIO_PIN_RESET); }
static void BMP280_CS_High(void) { HAL_GPIO_WritePin(BMP280_CS_PORT, BMP280_CS_PIN, GPIO_PIN_SET); }

static uint8_t BMP280_ReadReg(uint8_t reg)
{
  uint8_t tx = reg | 0x80;  /* bit 7 = 1 para lectura */
  uint8_t rx;
  BMP280_CS_Low();
  HAL_SPI_TransmitReceive(&hspi1, &tx, &rx, 1, HAL_MAX_DELAY);
  HAL_SPI_Receive(&hspi1, &rx, 1, HAL_MAX_DELAY);
  BMP280_CS_High();
  return rx;
}

static void BMP280_ReadRegs(uint8_t reg, uint8_t* buf, uint8_t len)
{
  uint8_t tx = reg | 0x80;
  BMP280_CS_Low();
  HAL_SPI_Transmit(&hspi1, &tx, 1, HAL_MAX_DELAY);
  HAL_SPI_Receive(&hspi1, buf, len, HAL_MAX_DELAY);
  BMP280_CS_High();
}

static void BMP280_WriteReg(uint8_t reg, uint8_t val)
{
  uint8_t tx[2] = {reg & 0x7F, val};  /* bit 7 = 0 para escritura */
  BMP280_CS_Low();
  HAL_SPI_Transmit(&hspi1, tx, 2, HAL_MAX_DELAY);
  BMP280_CS_High();
}

uint8_t BMP280_ReadID(void)
{
  return BMP280_ReadReg(BMP280_REG_ID);
  /* Debe devolver 0x58 */
}

Codigo HAL: Calibracion y lectura compensada

C — BMP280_Calibration y compensacion
/* Parametros de calibracion del BMP280 */
static uint16_t dig_T1;
static int16_t  dig_T2, dig_T3;
static uint16_t dig_P1;
static int16_t  dig_P2, dig_P3, dig_P4, dig_P5, dig_P6, dig_P7, dig_P8, dig_P9;
static int32_t  t_fine;

void BMP280_ReadCalibration(void)
{
  uint8_t calib[24];
  BMP280_ReadRegs(BMP280_REG_CALIB, calib, 24);

  dig_T1 = (uint16_t)((calib[1] << 8) | calib[0]);
  dig_T2 = (int16_t) ((calib[3] << 8) | calib[2]);
  dig_T3 = (int16_t) ((calib[5] << 8) | calib[4]);
  dig_P1 = (uint16_t)((calib[7] << 8) | calib[6]);
  dig_P2 = (int16_t) ((calib[9] << 8) | calib[8]);
  dig_P3 = (int16_t) ((calib[11] << 8) | calib[10]);
  dig_P4 = (int16_t) ((calib[13] << 8) | calib[12]);
  dig_P5 = (int16_t) ((calib[15] << 8) | calib[14]);
  dig_P6 = (int16_t) ((calib[17] << 8) | calib[16]);
  dig_P7 = (int16_t) ((calib[19] << 8) | calib[18]);
  dig_P8 = (int16_t) ((calib[21] << 8) | calib[20]);
  dig_P9 = (int16_t) ((calib[23] << 8) | calib[22]);
}

void BMP280_Configure(void)
{
  /* Modo normal, oversampling x1 temperatura, x1 presion */
  BMP280_WriteReg(BMP280_REG_CTRL, 0x27);   /* osrs_t=1, osrs_p=1, mode=normal */
  BMP280_WriteReg(BMP280_REG_CONFIG, 0xA0); /* t_sb=1000ms, filter=off */
}

/* Formula de compensacion de temperatura del datasheet Bosch */
static int32_t BMP280_CompensateTemp(int32_t adc_T)
{
  int32_t var1, var2, T;
  var1 = ((((adc_T >> 3) - ((int32_t)dig_T1 << 1))) * ((int32_t)dig_T2)) >> 11;
  var2 = (((((adc_T >> 4) - ((int32_t)dig_T1)) *
            ((adc_T >> 4) - ((int32_t)dig_T1))) >> 12) * ((int32_t)dig_T3)) >> 14;
  t_fine = var1 + var2;
  T = (t_fine * 5 + 128) >> 8;
  return T;  /* Temperatura en centesimas de C (ej: 2515 = 25.15 C) */
}

/* Formula de compensacion de presion del datasheet Bosch */
static uint32_t BMP280_CompensatePress(int32_t adc_P)
{
  int64_t var1, var2, p;
  var1 = ((int64_t)t_fine) - 128000;
  var2 = var1 * var1 * (int64_t)dig_P6;
  var2 = var2 + ((var1 * (int64_t)dig_P5) << 17);
  var2 = var2 + (((int64_t)dig_P4) << 35);
  var1 = ((var1 * var1 * (int64_t)dig_P3) >> 8) +
         ((var1 * (int64_t)dig_P2) << 12);
  var1 = (((((int64_t)1) << 47) + var1)) * ((int64_t)dig_P1) >> 33;

  if (var1 == 0) return 0;

  p = 1048576 - adc_P;
  p = (((p << 31) - var2) * 3125) / var1;
  var1 = (((int64_t)dig_P9) * (p >> 13) * (p >> 13)) >> 25;
  var2 = (((int64_t)dig_P8) * p) >> 19;
  p = ((p + var1 + var2) >> 8) + (((int64_t)dig_P7) << 4);

  return (uint32_t)(p / 256);  /* Presion en Pa (divide por 100 para hPa) */
}

Codigo HAL: Lectura completa de presion y temperatura

C — BMP280_ReadData
#include <math.h>

void BMP280_ReadData(float* temperature, float* pressure_hPa, float* altitude)
{
  uint8_t buf[6];
  int32_t adc_T, adc_P;

  /* Lee temperatura y presion crudos */
  BMP280_ReadRegs(BMP280_REG_PRESS, buf, 6);

  adc_P = ((int32_t)buf[0] << 12) | ((int32_t)buf[1] << 4) | (buf[2] >> 4);
  adc_T = ((int32_t)buf[3] << 12) | ((int32_t)buf[4] << 4) | (buf[5] >> 4);

  int32_t temp_centic = BMP280_CompensateTemp(adc_T);
  *temperature = temp_centic / 100.0f;

  uint32_t press_Pa = BMP280_CompensatePress(adc_P);
  *pressure_hPa = press_Pa / 100.0f;

  /* Formula barometrica para altitud (ISA) */
  /* h = 44330 * (1 - (P/P0)^(1/5.255)) */
  /* P0 = 1013.25 hPa (presion al nivel del mar) */
  *altitude = 44330.0f * (1.0f - powf(*pressure_hPa / 1013.25f, 0.1903f));
}

03 ADC: Sensor Industrial 4-20mA

Senal 4-20mA (Estandar Industrial)

El lazo de corriente 4-20mA es el estandar universal para transmisores industriales. La corriente se mantiene constante independientemente de la resistencia del cable, lo que lo hace inmune al ruido electromagnetico y a las caidas de voltaje en largas distancias.

  • 4 mA — representa el valor minimo (zero live, permite detectar cable roto)
  • 20 mA — representa el valor maximo del rango
  • < 4 mA — indica fallo (cable roto)
  • > 20 mA — indica corto o sobre-rango

Circuito de acondicionamiento

Para leer 4-20mA con un ADC de voltaje, se usa un resistor shunt de 250 que convierte la corriente en voltaje:

  • 4 mA * 250 = 1.0 V (minimo)
  • 20 mA * 250 = 5.0 V (maximo)

Como el ADC del STM32N6 opera a 3.3V, necesitamos un divisor resistivo o un resistor de valor menor. Usamos un divisor 3.3V/5V para escalar el rango:

Diagrama del circuito completo

Circuito de acondicionamiento 4-20mA +24V +3.3V (ref ADC) | | [Sensor] [250] <--- Resistor shunt | (4-20mA) | +--------------------+-----> [Divisor 3.3V/5V] -----> PA0 (ADC1_IN0) | | GND GND Divisor resistivo (escala 5V a 3.3V): Vout = Vin * (R2 / (R1 + R2)) con R1=10k, R2=20k => ratio = 0.667 5.0V * 0.667 = 3.33V (dentro del rango del ADC) STM32N6 Circuito externo PA0(ADC1_IN0)<-------Vout del divisor 3.3V----------------VCC sensor / pull-ups GND-----------------GND comun
Nota practica: En produccion industrial se usa un resistor de precision de 0.1% (250) y un amplificador de instrumentacion para aislamiento galvanico. En este modulo simplificamos con un divisor resistivo.

Codigo HAL: Lectura y conversion a presion

C — Sensor4_20_Read
#include <stdint.h>

/* Parametros del sistema */
#define ADC_RESOLUTION    4095.0f   /* 12 bits */
#define ADC_VREF          3.3f      /* Voltaje de referencia */
#define V_MIN             1.0f      /* 4mA * 250 */
#define V_MAX             5.0f      /* 20mA * 250 */
#define P_MIN             0.0f      /* PSI minimo */
#define P_MAX             100.0f    /* PSI maximo (rango del sensor) */
#define DIVISOR_RATIO     0.667f    /* 3.3V/5V = 0.667 */

/* Parametros de calibracion (ajustables) */
static float cal_offset = 0.0f;
static float cal_gain   = 1.0f;

uint16_t ADC_ReadRaw(void)
{
  HAL_ADC_Start(&hadc1);
  HAL_ADC_PollForConversion(&hadc1, HAL_MAX_DELAY);
  uint16_t raw = HAL_ADC_GetValue(&hadc1);
  HAL_ADC_Stop(&hadc1);
  return raw;
}

/* Filtro de promedio movil de 16 muestras */
#define FILTER_SIZE 16

float ADC_ReadFiltered(void)
{
  static uint16_t buffer[FILTER_SIZE] = {0};
  static uint8_t idx = 0;
  static uint8_t filled = 0;
  uint32_t sum = 0;

  buffer[idx] = ADC_ReadRaw();
  idx = (idx + 1) % FILTER_SIZE;
  if (idx == 0) filled = 1;

  uint8_t count = filled ? FILTER_SIZE : idx;
  for (uint8_t i = 0; i < count; i++)
    sum += buffer[i];

  return (float)sum / count;
}

float RawToVoltage(uint16_t raw)
{
  /* Convierte raw a voltaje en el ADC (0-3.3V) */
  float v_adc = (raw / ADC_RESOLUTION) * ADC_VREF;
  /* Compensa el divisor resistivo para obtener el voltaje real */
  float v_real = v_adc / DIVISOR_RATIO;
  return v_real;
}

float VoltageToPressure(float voltage)
{
  /* Ecuacion lineal: P = ((V - Vmin) / (Vmax - Vmin)) * (Pmax - Pmin) + Pmin */
  if (voltage < V_MIN) voltage = V_MIN;
  if (voltage > V_MAX) voltage = V_MAX;

  float pressure = ((voltage - V_MIN) / (V_MAX - V_MIN)) * (P_MAX - P_MIN) + P_MIN;

  /* Aplica calibracion */
  pressure = (pressure + cal_offset) * cal_gain;

  return pressure;
}

float Sensor_ReadPressure(void)
{
  float raw_avg = ADC_ReadFiltered();
  float voltage = RawToVoltage((uint16_t)raw_avg);
  float pressure = VoltageToPressure(voltage);
  return pressure;
}

/* Funcion de calibracion: ajusta offset con sensor en valor conocido */
void Sensor_CalibrateOffset(float known_pressure_psi)
{
  float raw_avg = ADC_ReadFiltered();
  float voltage = RawToVoltage((uint16_t)raw_avg);
  float measured = VoltageToPressure(voltage);
  cal_offset = known_pressure_psi - measured;
}

/* Funcion de calibracion: ajusta ganancia con dos puntos */
void Sensor_CalibrateGain(float p1_measured, float p1_actual,
                           float p2_measured, float p2_actual)
{
  if (p2_measured != p1_measured)
    cal_gain = (p2_actual - p1_actual) / (p2_measured - p1_measured);
}

Codigo HAL: Deteccion de fallos

C — Sensor4_20_CheckFault
typedef enum {
  SENSOR_OK = 0,
  SENSOR_FAULT_OPEN,     /* Cable roto: < 4mA */
  SENSOR_FAULT_SHORT,    /* Sensor en corto: > 20mA */
  SENSOR_FAULT_OVERRANGE /* Fuera de rango */
} SensorStatus_t;

#define I_MIN_mA    4.0f
#define I_MAX_mA    20.0f
#define I_FAULT_LOW 3.5f   /* Umbral para cable roto */
#define I_FAULT_HIGH 22.0f /* Umbral para corto */

/* Convierte voltaje a corriente estimada */
float VoltageToCurrent(float voltage)
{
  /* I = V / R_shunt */
  return (voltage / 250.0f) * 1000.0f; /* en mA */
}

SensorStatus_t Sensor_CheckFault(float* out_current_mA)
{
  float raw_avg = ADC_ReadFiltered();
  float v_adc = (raw_avg / ADC_RESOLUTION) * ADC_VREF;
  float v_real = v_adc / DIVISOR_RATIO;
  float current_mA = VoltageToCurrent(v_real);

  if (out_current_mA) *out_current_mA = current_mA;

  if (current_mA < I_FAULT_LOW)
    return SENSOR_FAULT_OPEN;   /* Cable roto o desconectado */

  if (current_mA > I_FAULT_HIGH)
    return SENSOR_FAULT_SHORT;  /* Sensor en corto */

  if (current_mA < I_MIN_mA - 0.5f || current_mA > I_MAX_mA + 0.5f)
    return SENSOR_FAULT_OVERRANGE;

  return SENSOR_OK;
}

const char* SensorStatusToString(SensorStatus_t status)
{
  switch (status)
  {
    case SENSOR_OK:            return "OK";
    case SENSOR_FAULT_OPEN:    return "FALLO: Cable roto";
    case SENSOR_FAULT_SHORT:   return "FALLO: Sensor en corto";
    case SENSOR_FAULT_OVERRANGE: return "FALLO: Fuera de rango";
    default:                   return "Desconocido";
  }
}

Main: Loop con monitoreo

C — main.c (ADC loop)
int main(void)
{
  HAL_Init();
  SystemClock_Config();
  MX_ADC1_Init();
  MX_USART2_UART_Init();

  printf("Sensor 4-20mA iniciado\r\n");

      printf("ALARMA: %s  Corriente: %.2f mA\r\n",
             SensorStatusToString(status), current_mA);
    }

    HAL_Delay(500);
  }
}

04 DMA para Adquisicion Continua

Que es DMA?

DMA (Direct Memory Access) es un controlador independiente de la CPU que transfiere datos directamente entre perifericos y memoria RAM sin intervencion del procesador. Esto libera la CPU para ejecutar otras tareas mientras los datos se adquieren en segundo plano.

  • Ventaja 1: La CPU no gasta ciclos en transferir datos byte a byte
  • Ventaja 2: Se pueden adquirir muestras a alta frecuencia sin perder datos
  • Ventaja 3: Modo circular permite adquisicion continua sin pausas

Diagrama de flujo DMA

Flujo de datos con DMA (sin intervencion de CPU) +--------+ +-----+ +-----+ +--------+ | Sensor |------->| ADC |------->| DMA |------->| RAM | | Analog | | 12b | | CH1 | | Buffer | +--------+ +-----+ +-----+ | 64 smpl| +--------+ | +----------------------+ | +--------+ +---------v--------+ | CPU |<-------| Callback (RMS) | |Procesa | | - Half Complete | | datos | | - Full Complete | +--------+ +------------------+ Modo Circular: Cuando el DMA llena el buffer, vuelve automaticamente al inicio sin detenerse.
Nota: En el STM32N6, el DMA tiene acceso a toda la memoria (incluida la SRAM externa) mediante su bus dedicado AHB. El ADC puede disparar el DMA automaticamente al final de cada conversion.

Codigo HAL: Inicializacion DMA

C — MX_DMA_Init
DMA_HandleTypeDef hdma_adc1;

void MX_DMA_Init(void)
{
  __HAL_RCC_DMA1_CLK_ENABLE();

  hdma_adc1.Instance = DMA1_Channel1;
  hdma_adc1.Init.Request = DMA_REQUEST_ADC1;
  hdma_adc1.Init.Direction = DMA_PERIPH_TO_MEMORY;
  hdma_adc1.Init.PeriphInc = DMA_PINC_DISABLE;
  hdma_adc1.Init.MemInc = DMA_MINC_ENABLE;
  hdma_adc1.Init.PeriphDataAlignment = DMA_PDATAALIGN_HALFWORD; /* 16 bits */
  hdma_adc1.Init.MemDataAlignment = DMA_MDATAALIGN_HALFWORD;
  hdma_adc1.Init.Mode = DMA_CIRCULAR;  /* Modo circular: buffer rotativo */
  hdma_adc1.Init.Priority = DMA_PRIORITY_HIGH;

  if (HAL_DMA_Init(&hdma_adc1) != HAL_OK)
  {
    Error_Handler();
  }

  /* Vincula el DMA al ADC */
  __HAL_LINKDMA(&hadc1, DMA_Handle, hdma_adc1);

  /* Habilita interrupciones del DMA */
  HAL_NVIC_SetPriority(DMA1_Channel1_IRQn, 0, 0);
  HAL_NVIC_EnableIRQ(DMA1_Channel1_IRQn);
}

void DMA1_Channel1_IRQHandler(void)
{
  HAL_DMA_IRQHandler(&hdma_adc1);
}

Codigo HAL: Callbacks y calculo de RMS

C — Callbacks DMA + RMS
#include <math.h>
#include <string.h>
#include <stdio.h>

static volatile uint8_t adc_half_complete = 0;
static volatile uint8_t adc_full_complete = 0;
static volatile float rms_value = 0.0f;

/* Se llama cuando la primera mitad del buffer esta llena */
void HAL_ADC_ConvHalfCpltCallback(ADC_HandleTypeDef* hadc)
{
  if (hadc->Instance == ADC1)
  {
    adc_half_complete = 1;
  }
}

/* Se llama cuando el buffer completo esta lleno */
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc)
{
  if (hadc->Instance == ADC1)
  {
    adc_full_complete = 1;
  }
}

/* Calcula RMS sobre un bloque de muestras */
float CalculateRMS(uint16_t* samples, uint16_t count)
{
  float sum_sq = 0.0f;
  float voltage;

  for (uint16_t i = 0; i < count; i++)
  {
    /* Convierte raw a voltaje */
    voltage = (samples[i] / 4095.0f) * 3.3f;
    sum_sq += voltage * voltage;
  }

  return sqrtf(sum_sq / count);
}

/* Procesa las muestras listas y opcionalmente las muestra */
void ProcessADCData(void)
{
  if (adc_half_complete)
  {
    adc_half_complete = 0;
    /* Procesa primera mitad (muestras 0 a 31) */
    float rms_first = CalculateRMS(&adc_buffer[0], ADC_BUFFER_SIZE / 2);
    printf("RMS[0-31]: %.3f V  ", rms_first);

    /* Muestra las ultimas 16 muestras en formato osciloscopio ASCII */
    PrintWaveform(&adc_buffer[0], ADC_BUFFER_SIZE / 2);
  }

  if (adc_full_complete)
  {
    adc_full_complete = 0;
    /* Procesa segunda mitad (muestras 32 a 63) */
    float rms_second = CalculateRMS(&adc_buffer[ADC_BUFFER_SIZE / 2], ADC_BUFFER_SIZE / 2);
    printf("RMS[32-63]: %.3f V\r\n", rms_second);

    rms_value = rms_second;  /* Valor RMS actualizado */
  }
}

/* Visualizador ASCII tipo osciloscopio */
void PrintWaveform(uint16_t* samples, uint16_t count)
{
  /* Muestra las ultimas 16 muestras como grafico ASCII */
  uint16_t display_count = (count > 16) ? 16 : count;
  char line[64];

  printf("Waveform: ");
  for (uint16_t i = 0; i < display_count; i++)
  {
    uint16_t idx = (count > 16) ? (count - 16 + i) : i;
    int bar = (int)((samples[idx] * 30) / 4095);  /* Escala a 0-30 */
    if (bar > 29) bar = 29;
    memset(line, ' ', 30);
    line[bar] = '*';
    line[30] = '\0';
    printf("|%s| ", line);
  }
  printf("\r\n");
}

Main: Adquisicion continua con DMA

C — main.c (DMA loop)
int main(void)
{
  HAL_Init();
  SystemClock_Config();
  MX_ADC1_DMA_Init();
  MX_USART2_UART_Init();

  printf("ADC + DMA continuo iniciado\r\n");
  printf("Buffer: %d muestras @ 12-bit\r\n\r\n", ADC_BUFFER_SIZE);

  /* Inicia la adquisicion continua via DMA */
  ADC_StartDMA();

  while (1)
  {
    ProcessADCData();

    /* Aqui la CPU esta libre para otras tareas */
    /* Por ejemplo: procesamiento de red, comunicacion, control */

    HAL_Delay(10);  /* Pequena pausa para no saturar UART */
  }
}

/* Handler de error */
void Error_Handler(void)
{
  __disable_irq();
  while (1)
  {
    HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_5);  /* LED de error */
    HAL_Delay(100);
  }
}
